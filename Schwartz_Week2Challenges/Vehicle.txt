public virtual class Vehicle {
    
    public enum TransportMode {AIR, WATER, LAND}
    public static integer totalVehicles = 0;
    public TransportMode transportType{get; set;}
    
    private integer speed = 0;
    
    public String Name {get; set;}
    
    public Vehicle(){
        totalVehicles++;
    }
    
    public void Movement(){
        System.debug('Vehicle is moving.');
    }
    
    public void Accelerate(){
        System.debug('Vehicle is accelerating');
        addSpeed(10);
    }
    
    public virtual void AddSpeed(integer sp){
        this.speed += sp;
    }
    
    public Integer GetSpeed(){
        return this.speed;
    }
    
    public void SetSpeed(integer i){
        if(i > 0)
            this.speed = i;
        else
            System.debug('Invalid speed param....:(');
    }
    
    public void DecreaseSpeed(integer decrease){
        this.speed -= decrease;
        if(this.speed < 0){
            speed = 0;
            System.debug('Speed is already 0. Stop Decelerating!!!');
        }
    }
    
    public void Stop(){
        System.debug('Vehicle has stopped');
    }
    
    public void print(String str){
        System.debug(str);
    }

}
